## 二、P0 优化方案

### 2.1 AudioWorklet 替换 ScriptProcessorNode

#### 问题分析

```typescript
// 当前实现 - ScriptProcessorNode 在主线程运行
const processor = audioContext.createScriptProcessor(4096, 1, 1);
processor.onaudioprocess = (e) => {
    // 这里的处理会阻塞主线程
    const inputData = e.inputBuffer.getChannelData(0);
    // ... 重采样、编码等
};
```

**问题**：
- `ScriptProcessorNode` 已被 W3C 标记为废弃
- 在主线程执行，高负载时会导致 UI 卡顿
- 4096 样本缓冲区带来 ~85ms 固定延迟

#### 解决方案

**Step 1**: 创建 AudioWorklet Processor

```javascript
// public/audio-worklet-processor.js
class AudioProcessor extends AudioWorkletProcessor {
    constructor() {
        super();
        this.bufferSize = 4096;
        this.buffer = new Float32Array(this.bufferSize);
        this.bufferIndex = 0;
    }

    process(inputs, outputs, parameters) {
        const input = inputs[0];
        if (!input || !input[0]) return true;

        const inputChannel = input[0];
        
        for (let i = 0; i < inputChannel.length; i++) {
            this.buffer[this.bufferIndex++] = inputChannel[i];
            
            if (this.bufferIndex >= this.bufferSize) {
                // 发送到主线程
                this.port.postMessage({
                    type: 'audio',
                    buffer: this.buffer.slice()
                });
                this.bufferIndex = 0;
            }
        }
        
        return true;
    }
}

registerProcessor('audio-processor', AudioProcessor);
```

**Step 2**: 更新 useAudioRecorder Hook


### 2.2 流式 TTS 播放

#### 问题分析

```python
# 当前实现 - 等待所有音频块后一次性发送
audio_chunks = []
async for chunk in result.value:
    audio_chunks.append(chunk)
audio_data = b"".join(audio_chunks)  # 等待完成
await self._send_tts_audio(audio_data)
```

**问题**：
- 长文本需要等待整个 TTS 合成完成
- 首字节延迟 = TTS 合成时间 + 网络传输时间

#### 解决方案

**Step 1**: 后端流式发送 TTS 块

```python
# backend/src/sales_bot/websocket/enhanced_handler.py

async def _send_tts_response_streaming(self, text: str):
    """流式发送 TTS 音频块"""
    tts_service = get_tts_service()
    trace_id = self.context.trace_id if self.context else None
    
    chunk_index = 0
    total_duration_ms = 0
    
    try:
        result = await tts_service.synthesize(text)
        
        if not result.is_success:
            await self._send_tts_fallback(text, trace_id)
            return
        
        # 流式发送每个音频块
        async for chunk in result.value:
            chunk_duration_ms = int(len(chunk) / 32)  # MP3 ~32 bytes/ms
            total_duration_ms += chunk_duration_ms
            
            await self.manager.send_json(
                self.websocket,
                {
                    "type": "tts_chunk",
                    "timestamp": datetime.utcnow().isoformat(),
                    "trace_id": trace_id,
                    "data": {
                        "chunk_index": chunk_index,
                        "audio": base64.b64encode(chunk).decode("utf-8"),
                        "duration_ms": chunk_duration_ms,
                        "is_final": False,
                    },
                },
            )
            chunk_index += 1
        
        # 发送结束标记
        await self.manager.send_json(
            self.websocket,
            {
                "type": "tts_chunk",
                "timestamp": datetime.utcnow().isoformat(),
                "trace_id": trace_id,
                "data": {
                    "text": text,
                    "chunk_index": chunk_index,
                    "audio": "",
                    "duration_ms": 0,
                    "is_final": True,
                    "total_duration_ms": total_duration_ms,
                },
            },
        )
        
    except Exception as e:
        logger.error(f"Streaming TTS error: {e}")
        await self._send_tts_fallback(text, trace_id)
```

**Step 2**: 前端使用 MediaSource API 流式播放

```typescript
// web/src/hooks/use-streaming-audio-player.ts

interface StreamingAudioPlayer {
    start: () => void;
    appendChunk: (chunk: ArrayBuffer) => void;
    end: () => void;
    stop: () => void;
}

export function useStreamingAudioPlayer(): StreamingAudioPlayer {
    const mediaSourceRef = useRef<MediaSource | null>(null);
    const sourceBufferRef = useRef<SourceBuffer | null>(null);
    const audioRef = useRef<HTMLAudioElement | null>(null);
    const chunkQueueRef = useRef<ArrayBuffer[]>([]);
    const isAppendingRef = useRef(false);

    const appendNextChunk = useCallback(() => {
        if (isAppendingRef.current || chunkQueueRef.current.length === 0) {
            return;
        }
        
        const sourceBuffer = sourceBufferRef.current;
        if (!sourceBuffer || sourceBuffer.updating) {
            return;
        }
        
        isAppendingRef.current = true;
        const chunk = chunkQueueRef.current.shift()!;
        
        try {
            sourceBuffer.appendBuffer(chunk);
        } catch (e) {
            console.error('Failed to append buffer:', e);
            isAppendingRef.current = false;
        }
    }, []);

    const start = useCallback(() => {
        const audio = new Audio();
        const mediaSource = new MediaSource();
        
        audio.src = URL.createObjectURL(mediaSource);
        audioRef.current = audio;
        mediaSourceRef.current = mediaSource;
        
        mediaSource.addEventListener('sourceopen', () => {
            try {
                const sourceBuffer = mediaSource.addSourceBuffer('audio/mpeg');
                sourceBufferRef.current = sourceBuffer;
                
                sourceBuffer.addEventListener('updateend', () => {
                    isAppendingRef.current = false;
                    appendNextChunk();
                });
                
                // 开始播放
                audio.play().catch(console.error);
            } catch (e) {
                console.error('Failed to create source buffer:', e);
            }
        });
    }, [appendNextChunk]);

    const appendChunk = useCallback((chunk: ArrayBuffer) => {
        chunkQueueRef.current.push(chunk);
        appendNextChunk();
    }, [appendNextChunk]);

    const end = useCallback(() => {
        const mediaSource = mediaSourceRef.current;
        const sourceBuffer = sourceBufferRef.current;
        
        if (mediaSource && sourceBuffer && !sourceBuffer.updating) {
            try {
                mediaSource.endOfStream();
            } catch (e) {
                console.error('Failed to end stream:', e);
            }
        }
    }, []);

    const stop = useCallback(() => {
        const audio = audioRef.current;
        if (audio) {
            audio.pause();
            audio.src = '';
        }
        chunkQueueRef.current = [];
        isAppendingRef.current = false;
    }, []);

    return { start, appendChunk, end, stop };
}
```

## 三、P1 优化方案

### 3.1 完善打断机制

#### 当前问题

```
用户开始说话 → 发送 interrupt → 后端停止 ASR → 但 TTS 仍在播放
```

#### 解决方案

**完整打断链路**：

```
用户开始说话
    ↓
前端检测到 (VAD 或按钮)
    ↓
立即停止 TTS 播放 (前端)
    ↓
发送 interrupt 消息
    ↓
后端停止当前处理
    ↓
后端发送 interrupted 确认
    ↓
前端切换到 listening 状态
```

**前端实现**：

```typescript
// web/src/hooks/use-practice-websocket.ts

const handleUserStartSpeaking = useCallback(() => {
    // 1. 立即停止 TTS 播放
    if (state.isPlayingAudio) {
        streamingPlayer.stop();
        window.speechSynthesis?.cancel();
        setState(prev => ({ ...prev, isPlayingAudio: false }));
    }
    
    // 2. 清空音频队列
    audioQueueRef.current = [];
    
    // 3. 发送打断信号
    sendMessage("interrupt", { 
        reason: "user_speaking",
        timestamp: Date.now()
    });
    
    // 4. 更新状态
    setState(prev => ({ ...prev, aiState: "listening" }));
}, [state.isPlayingAudio, streamingPlayer, sendMessage]);
```

**后端实现**：

```python
# backend/src/sales_bot/websocket/enhanced_handler.py

async def _handle_interrupt(self, data: dict):
    """处理用户打断"""
    reason = data.get("reason", "unknown")
    logger.info(f"User interrupted: {reason}")
    
    # 1. 停止 ASR
    await self._stop_streaming_asr()
    
    # 2. 取消正在进行的 LLM 请求
    if self._llm_task and not self._llm_task.done():
        self._llm_task.cancel()
    
    # 3. 停止 TTS 生成
    if self._tts_task and not self._tts_task.done():
        self._tts_task.cancel()
    
    # 4. 发送确认
    await self.manager.send_json(
        self.websocket,
        {
            "type": "interrupted",
            "timestamp": datetime.utcnow().isoformat(),
            "data": {"reason": reason},
        },
    )
    
    # 5. 切换到监听状态
    await self._send_status("listening")
```

### 3.2 背压控制机制

#### 当前问题

```python
# 无界队列，可能无限增长
self.asr_queue = asyncio.Queue()
await self.asr_queue.put(audio_bytes)  # 永不阻塞
```

#### 解决方案

```python
# backend/src/sales_bot/websocket/enhanced_handler.py

class EnhancedSalesHandler(BaseWebSocketHandler):
    # 队列配置
    ASR_QUEUE_MAX_SIZE = 100  # 最大 100 个音频块
    ASR_QUEUE_HIGH_WATERMARK = 80  # 80% 时发送警告
    
    async def _start_streaming_asr(self):
        """启动流式 ASR，使用有界队列"""
        self.asr_queue = asyncio.Queue(maxsize=self.ASR_QUEUE_MAX_SIZE)
        self.asr_task = asyncio.create_task(self._run_streaming_asr())
        
    async def _handle_audio_chunk(self, data: dict):
        """处理音频块，带背压控制"""
        audio_base64 = data.get("audio", "")
        
        if not audio_base64 or not self.asr_queue:
            return
            
        try:
            audio_bytes = base64.b64decode(audio_base64)
            
            # 检查队列水位
            queue_size = self.asr_queue.qsize()
            
            if queue_size >= self.ASR_QUEUE_HIGH_WATERMARK:
                # 发送背压警告
                await self._send_backpressure_warning(queue_size)
            
            # 尝试放入队列，超时则丢弃
            try:
                await asyncio.wait_for(
                    self.asr_queue.put(audio_bytes),
                    timeout=0.1
                )
            except asyncio.TimeoutError:
                logger.warning("ASR queue full, dropping audio chunk")
                await self._send_backpressure_signal()
                
        except Exception as e:
            logger.error(f"Failed to process audio chunk: {e}")
    
    async def _send_backpressure_warning(self, queue_size: int):
        """发送背压警告"""
        await self.manager.send_json(
            self.websocket,
            {
                "type": "backpressure",
                "timestamp": datetime.utcnow().isoformat(),
                "data": {
                    "level": "warning",
                    "queue_size": queue_size,
                    "max_size": self.ASR_QUEUE_MAX_SIZE,
                },
            },
        )
    
    async def _send_backpressure_signal(self):
        """发送背压信号，通知前端降速"""
        await self.manager.send_json(
            self.websocket,
            {
                "type": "backpressure",
                "timestamp": datetime.utcnow().isoformat(),
                "data": {
                    "level": "critical",
                    "action": "slow_down",
                },
            },
        )
```

**前端响应背压**：

```typescript
case "backpressure": {
    const data = message.data;
    
    if (data.level === "critical") {
        // 降低发送频率
        setAudioSendInterval(prev => Math.min(prev * 2, 200));
        console.warn("Server backpressure, slowing down audio send");
    } else if (data.level === "warning") {
        console.log(`Server queue at ${data.queue_size}/${data.max_size}`);
    }
    break;
}
```

### 4.1 高质量重采样

#### 当前问题

```typescript
// 线性插值，48kHz → 16kHz 会产生混叠
output[i] = inputData[srcIndexFloor] * (1 - t) + inputData[srcIndexCeil] * t;
```

#### 解决方案：使用 OfflineAudioContext

```typescript
async function highQualityResample(
    inputBuffer: Float32Array,
    inputSampleRate: number,
    outputSampleRate: number
): Promise<Float32Array> {
    const duration = inputBuffer.length / inputSampleRate;
    const outputLength = Math.ceil(duration * outputSampleRate);
    
    // 创建离线上下文
    const offlineCtx = new OfflineAudioContext(
        1,
        outputLength,
        outputSampleRate
    );
    
    // 创建输入缓冲区
    const audioBuffer = offlineCtx.createBuffer(
        1,
        inputBuffer.length,
        inputSampleRate
    );
    audioBuffer.getChannelData(0).set(inputBuffer);
    
    // 创建源节点
    const source = offlineCtx.createBufferSource();
    source.buffer = audioBuffer;
    source.connect(offlineCtx.destination);
    source.start();
    
    // 渲染
    const renderedBuffer = await offlineCtx.startRendering();
    return renderedBuffer.getChannelData(0);
}
```

### 4.2 二进制传输替代 Base64

#### 当前问题

```typescript
// Base64 编码增加 33% 数据量
const base64 = btoa(String.fromCharCode(...bytes));
```

#### 解决方案：WebSocket 二进制帧

```typescript
// 前端：发送二进制数据
const sendAudioBinary = useCallback((pcmData: Int16Array) => {
    if (wsRef.current?.readyState === WebSocket.OPEN) {
        // 添加消息头 (4 bytes: type + sequence)
        const header = new Uint8Array(4);
        header[0] = 0x01;  // type: audio
        header[1] = (sequenceRef.current >> 16) & 0xFF;
        header[2] = (sequenceRef.current >> 8) & 0xFF;
        header[3] = sequenceRef.current & 0xFF;
        sequenceRef.current++;
        
        // 合并头和数据
        const message = new Uint8Array(header.length + pcmData.byteLength);
        message.set(header);
        message.set(new Uint8Array(pcmData.buffer), header.length);
        
        wsRef.current.send(message);
    }
}, []);
```

```python
# 后端：处理二进制消息
async def handle_binary_message(self, data: bytes):
    """处理二进制 WebSocket 消息"""
    if len(data) < 4:
        return
    
    msg_type = data[0]
    sequence = (data[1] << 16) | (data[2] << 8) | data[3]
    payload = data[4:]
    
    if msg_type == 0x01:  # audio
        await self._handle_audio_binary(payload, sequence)
```
